function get_method_id(method) result(id)

   !> Name of the method to look up
   character(len=*), intent(in) :: method

   character(len=len(method)) :: lc_method

   integer :: id
   integer :: i, j

   j = 0
   do i = 1, len(method)
      if (method(i:i) /= "-") then
